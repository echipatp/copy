  <script>

    const languages = ["RO", "ES", "EN", "DE", "FR", "PT", "IT"];

    const params = new URLSearchParams(window.location.search);


    const container = document.getElementById("container");


    languages.forEach(lang => {

      const text = params.get(lang.toLowerCase()) || `Citat ${lang} indisponibil.`;


      const box = document.createElement("div");

      box.className = "citat-box";


      box.innerHTML = `

        <label><strong>Citat ${lang}:</strong></label>

        <textarea id="citat-${lang}" readonly>${text}</textarea>

        <button onclick="copiere('${lang}')">Copiază ${lang}</button>

      `;


      container.appendChild(box);

    });


    function copiere(lang) {

      const textarea = document.getElementById(`citat-${lang}`);

      textarea.select();

      textarea.setSelectionRange(0, 99999);

      document.execCommand("copy");

      alert(`Citatul ${lang} a fost copiat!`);

    }

  </script>
